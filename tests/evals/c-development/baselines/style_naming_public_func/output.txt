#include <stdint.h>
#include <stddef.h>

typedef struct my_utils_list_node_t {
    void*                        data;
    struct my_utils_list_node_t* next;
} my_utils_list_node_t;

typedef struct {
    my_utils_list_node_t* head;
    size_t                size;
} my_utils_list_t;

void    my_utils_list_init(my_utils_list_t* list);
void    my_utils_list_deinit(my_utils_list_t* list);
int32_t my_utils_list_insert(my_utils_list_t* list, void* data);
size_t  my_utils_list_size(const my_utils_list_t* list);
